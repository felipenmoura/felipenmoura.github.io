Arrays

// .from()
function someFunc() {
    // Array.prototype.slice.call(arguments);
    return Array.from(arguments);
}

// .of()
ar1= Array(1, 2, 3); // [1, 2, 3]
ar2= Array(1); // [undefined]
ar3= Array.of(1); // [1]

// .fill()
[1, 2, 3].fill('X');         // ['X', 'X', 'X']
[1, 2, 3, 4, 5, 6].fill('X', 1, 3);         // [1, 'X', 'X', '4', 5, 6]

// .find() / .findIndex()
function is3or5(element, index, array){
    if(element == 3 || element == 5){
        return true;
    }else{
        return false;
    }
}
alert([4, 5, 6, 7].find( is3or5 ));
alert([1, 2, 3, 4].find( is3or5 ));

// .entries()
var arr = ['a', 'b', 'c'];
var eArr = arr.entries();

alert(eArr.next().value); // [0, 'a']
alert(eArr.next().value); // [1, 'b']
alert(eArr.next().value); // [2, 'c']
